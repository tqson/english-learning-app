JavaScript Cơ Bản

1. Kiểu dữ liệu (Data Types)

// Number let age = 25; let price = 99.99; // String let name = "John Doe"; let message = 'Hello World'; // Boolean let isActive = true; let isLoggedIn = false; // Null let empty = null; // Undefined let notDefined; // Object let person = { name: "John", age: 30, isStudent: false }; // Array let colors = ["red", "green", "blue"]; let numbers = [1, 2, 3, 4, 5]; // Symbol let sym = Symbol("description");

2. Biến và Phạm vi (Variables & Scope)

// Global scope var globalVar = "I'm global"; // Function scope function test() { var functionVar = "I'm function scoped"; } // Block scope { let blockVar = "I'm block scoped"; const constantVar = "I can't be changed"; }

3. Toán tử (Operators)

// Arithmetic let sum = 5 + 3; // 8 let diff = 10 - 4; // 6 let product = 3 * 4; // 12 let quotient = 15 / 3;// 5 // Comparison let isEqual = 5 === "5"; // false let isNotEqual = 5 !== "5"; // true let greaterThan = 10 > 5; // true let lessThan = 3 < 7; // true // Logical let and = true && false; // false let or = true || false; // true let not = !true; // false

4. Điều kiện (Conditions)

// If-else let score = 85; if (score >= 90) { console.log("Excellent"); } else if (score >= 80) { console.log("Good"); } else { console.log("Try harder"); } // Switch let day = "Monday"; switch (day) { case "Monday": console.log("Start of week"); break; case "Friday": console.log("Weekend coming"); break; default: console.log("Regular day"); }

5. Vòng lặp (Loops)

// For loop for (let i = 0; i < 5; i++) { console.log(i); } // While loop let count = 0; while (count < 3) { console.log(count); count++; } // For...of (arrays) let fruits = ["apple", "banana", "orange"]; for (let fruit of fruits) { console.log(fruit); } // For...in (objects) let car = { brand: "Toyota", model: "Camry", year: 2022 }; for (let key in car) { console.log(`${key}: ${car[key]}`); }

6. Functions

// Regular function function greet(name) { return `Hello, ${name}!`; } // Arrow function const multiply = (a, b) => a * b; // Function expression const add = function(x, y) { return x + y; }; // Callback function setTimeout(() => { console.log("Delayed message"); }, 1000);
